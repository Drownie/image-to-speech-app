import { StatusBar } from 'expo-status-bar';
import { StyleSheet, Text, View, Button, Image } from 'react-native';
import * as ImagePicker from 'expo-image-picker';
import Tesseract from 'tesseract.js';
import * as Speech from 'expo-speech';
import { useEffect, useState } from 'react';

export default function App() {
  const [image, setImage] = useState(null);
  const [text, setText] = useState(null);

  const pickImage = async () => {
    let result = await ImagePicker.launchImageLibraryAsync({
      mediaTypes: ImagePicker.MediaTypeOptions.All,
      allowsEditing: true,
      aspect: [4, 3],
      quality: 1,
    });

    console.log(result);

    if (!result.canceled) {
      setImage(result.assets[0]);
      extractText(result.assets[0]);
    }
  }

  const extractText = async (imageInput) => {
    Tesseract.recognize(
      imageInput.uri, 
      'eng', 
      { logger: m => console.log(m) }
      ).then(({ data: { text, confidence } }) => {
        console.log(text);
        console.log(confidence);
        setText(text);
        Speech.speak(text, {
          language: 'en',
          rate: .8,
        });
        
        // Speech.getAvailableVoicesAsync()
        //   .then((e) => console.log(e));
      })
  };

  return (
    <View style={styles.container}>
      <Text style={styles.textStyle}>Simple OCR APP</Text>
      <View>
        <Button title="Pick an image" onPress={pickImage} />
        <Button title="Extract" onPress={pickImage} />
      </View>
      {image && <Image source={{ uri: image.uri }} style={{ width: image.width, height: image.height, margin: 10 }} />}
      {text && <Text style={{ width: image.width, height: image.height, margin: 10 }}>{text}</Text>}
      <StatusBar style="auto" />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    backgroundColor: '#fff',
    alignItems: 'center',
    justifyContent: 'center',
    borderWidth: 4,
    borderColor: "#000",
  },
});
